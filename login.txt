<script>
  const debugEl = document.getElementById('debug');
  function showDebug(msg){
    debugEl.style.display='block';
    const now = new Date().toLocaleTimeString();
    debugEl.textContent = `[${now}] ` + msg + "\n" + debugEl.textContent;
    console.log(msg);
  }

  document.getElementById('doLogin').addEventListener('click', () => {
    const u = document.getElementById('uid').value.trim();
    const p = document.getElementById('pw').value;
    if(!u || !p){
      showDebug('Login: Missing username or password.');
      alert('Please enter username/email/phone and password.');
      return;
    }
    const masked = '•'.repeat(Math.max(6, p.length));
    showDebug(`Login attempt — username: "${u}", password: "${masked}" (masked)`);
    const success = Math.random() > 0.25;
    if(success){
      showDebug('Login simulated: SUCCESS');
      alert('Login simulated: Success!');
    } else {
      showDebug('Login simulated: FAILED (invalid credentials)');
      alert('Login simulated: Invalid credentials.');
    }
  });

  document.getElementById('oneTime').addEventListener('click', () => {
    const u = document.getElementById('uid').value.trim();
    const which = u ? `for ${u}` : '';
    showDebug(`Requested one-time code ${which}. (Simulated)`);
    alert('Simulated: One-time code emailed (check debug).');
  });

  document.getElementById('otherDevice').addEventListener('click', () => {
    showDebug('User chose "Use Another Device" — show instructions (simulated).');
    alert('Simulated: A prompt to use another device would appear here.');
  });

  document.getElementById('signupLink').addEventListener('click', (e) => {
    e.preventDefault();
    showDebug('User clicked Sign Up (simulated).');
    alert('Simulated: Redirecting to Sign Up page (not implemented).');
  });
</script>
